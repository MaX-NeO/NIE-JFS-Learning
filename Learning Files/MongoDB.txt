db.dropDatabase()


//Collections

show collections

// Create collection
db.createCollection("users")

// Drop collection
db.users.drop()

// Document
// Insert one document
db.users.insertOne({ name: "Alice", age: 25, city: "Delhi" })

// Insert many documents
db.users.insertMany([
  { name: "Bob", age: 30, city: "Mumbai" },
  { name: "Charlie", age: 28, city: "Chennai" }
])

// Find all documents
db.users.find()

// Find with condition
db.users.find({ city: "Mumbai" })

// Find one document
db.users.findOne({ name: "Alice" })

// Update one document
db.users.updateOne(
  { name: "Alice" },
  { $set: { age: 26 } }
)



// Delete one document
db.users.deleteOne({ name: "Bob" })
